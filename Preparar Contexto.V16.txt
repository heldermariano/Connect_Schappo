function getItemNome(numero) {
  const item = ITENS_PADRAO.find(i => i.numero === numero);
  return item ? item.nome : `Item ${numero}`;
}

function montarListaItensSimples() {
  let lista = '*📦 Equipamentos:*\n';
  ITENS_PADRAO.filter(i => i.categoria === 'equipamento').forEach(item => {
    lista += `  ${item.numero}. ${item.nome}\n`;
  });
  lista += '*🔧 Acessórios:*\n';
  ITENS_PADRAO.filter(i => i.categoria === 'acessorio').forEach(item => {
    lista += `  ${item.numero}. ${item.nome}\n`;
  });
  return lista.trim();
}

function formatarItensSelecionados(itens) {
  return itens.map(num => {
    const item = ITENS_PADRAO.find(i => i.numero === num);
    return item ? `✅ ${num}. ${item.nome}` : `✅ ${num}. Item`;
  }).join('\n');
}

function parsearNumeros(texto) {
  const matches = texto.match(/\d+/g);
  if (!matches) return [];
  return [...new Set(matches.map(n => parseInt(n)).filter(n => n >= 1 && n <= 13))].sort((a, b) => a - b);
}

function cancelarSessoesPendentes() {
  const sqlCancel = `
    UPDATE eeg_sessao_retirada 
    SET status = 'cancelada', atualizada_em = NOW()
    WHERE telefone LIKE '%' || RIGHT('${telefone}', 11) || '%'
      AND status NOT IN ('finalizada', 'cancelada', 'expirada')
  `;
  
  // Se já tem SQL pendente, adiciona como CTE
  if (executarSQLAntes && sqlQueryAntes) {
    sqlQueryAntes = `WITH _cancela_sessoes AS (${sqlCancel}) ${sqlQueryAntes}`;
  } else if (!executarSQLAntes) {
    // Marca para executar o cancel standalone
    executarSQLAntes = true;
    sqlAcaoAntes = 'cancelar_sessoes_pendentes';
    sqlQueryAntes = sqlCancel + ' RETURNING id';
  }
}

}

}

function handlePossivelDescricao() {
  const temProblemaPendente = sessaoData?.problema_pendente?.item_numero && 
                              sessaoData?.problema_pendente?.tipo_problema;
  
  if (temProblemaPendente) {
    tipoInput = 'possivel_descricao';
    dadosExtraidos.textoDigitado = content;
    
    executarSQLAntes = true;
    sqlAcaoAntes = 'registrar_problema_descricao';
    sqlQueryAntes = `
      WITH problema AS (
        SELECT 
          s.caixa_codigo,
          (s.problema_pendente->>'item_numero')::INTEGER as item_numero,
          (s.problema_pendente->>'tipo_problema')::VARCHAR as tipo_problema,
          (s.problema_pendente->>'origem')::VARCHAR as origem
        FROM eeg_sessao_retirada s
        WHERE s.telefone = '${telefone}'
          AND s.problema_pendente IS NOT NULL
          AND s.problema_pendente != '{}'::JSONB
          AND s.status NOT IN ('finalizada', 'cancelada', 'expirada')
        ORDER BY s.id DESC
        LIMIT 1
      ),
      registro AS (
        INSERT INTO eeg_problemas (caixa_codigo, caixa_id, item_numero, tipo_problema, descricao, reportado_por, criado_em)
        SELECT 
          p.caixa_codigo,
          (SELECT id FROM eeg_caixas WHERE UPPER(codigo) = UPPER(p.caixa_codigo)),
          p.item_numero,
          p.tipo_problema,
          '${content.replace(/'/g, "''")}',
          ${userId},
          NOW()
        FROM problema p
        WHERE p.caixa_codigo IS NOT NULL
        RETURNING id, caixa_codigo, item_numero
      ),
      limpa AS (
        UPDATE eeg_sessao_retirada 
        SET problema_pendente = NULL 
        WHERE telefone = '${telefone}' 
          AND problema_pendente IS NOT NULL
          AND EXISTS (SELECT 1 FROM registro)
      )
      SELECT 
        CASE WHEN r.id IS NOT NULL THEN true ELSE false END as sucesso,
        CASE WHEN r.id IS NOT NULL THEN 'Problema registrado' ELSE 'Nenhum problema pendente' END as mensagem,
        r.id as problema_id,
        COALESCE(r.caixa_codigo, '') as caixa_codigo,
        COALESCE(r.item_numero, 0) as item_numero,
        COALESCE((SELECT nome FROM eeg_itens_padrao WHERE numero = r.item_numero), '') as item_nome,
        COALESCE(p.tipo_problema, '') as tipo_problema,
        COALESCE(p.origem, 'reportar') as origem
      FROM problema p
      LEFT JOIN registro r ON true
    `;
    
    respostaJSON = {
      mensagem: "✅ *Problema registrado com sucesso!*\n\nCarregando...",
      botoes: [
        { id: "menu_principal", title: "🏠 Menu" },
        { id: "modulo_eeg", title: "🔙 Menu EEG" }
      ],
      _processarRegistroProblema: true,
      _descricao: content
    };
  } else {
    tipoInput = 'texto_livre';
    respostaJSON = {
      mensagem: `🤔 Não entendi "*${content.substring(0, 30)}...*".\n\nSelecione uma opção:`,
      botoes: [
        { id: "modulo_eeg", title: "🧠 Módulo EEG" },
        { id: "menu_principal", title: "🏠 Menu" }
      ]
    };
  }
}

function handleRetiradaModemNao(buttonId) {
  const caixa = buttonId.replace('modem_nao_', '').toUpperCase();
  dadosExtraidos.caixa = caixa;
  dadosExtraidos.modem = null;
  
  executarSQLAntes = true;
  sqlAcaoAntes = 'buscar_itens_resumo';
  sqlQueryAntes = `
    WITH retirada AS (
      SELECT * FROM buscar_itens_retirada_whatsapp(${userId}, '${caixa}')
    ),
    problemas AS (
      SELECT DISTINCT ON (p.item_numero) p.id, p.item_numero, ip.nome as item_nome, p.tipo_problema, p.descricao
      FROM eeg_problemas p
      JOIN eeg_itens_padrao ip ON ip.numero = p.item_numero
      WHERE UPPER(p.caixa_codigo) = '${caixa}' AND p.status = 'pendente'
      ORDER BY p.item_numero, p.id DESC
    ),
    notas AS (
      SELECT string_agg(n.nota || ' (' || n.supervisor_nome || ')', E'\\n' ORDER BY n.criado_em DESC) as texto,
             COUNT(*)::integer as qtd
      FROM eeg_notas_manutencao n
      WHERE n.caixa_codigo = '${caixa}' AND n.ativa = true
    )
    SELECT 
      r.sucesso, r.tem_itens, r.todos_itens, r.lista_texto, r.itens_array,
      COALESCE((SELECT COUNT(*) FROM problemas), 0)::integer as qtd_problemas,
      (SELECT string_agg(
        CASE p.tipo_problema
          WHEN 'defeito' THEN '🔴 ' || p.item_numero || '. ' || p.item_nome || ' (defeito - ' || COALESCE(p.descricao, '') || ')'
          WHEN 'faltando' THEN '🟡 ' || p.item_numero || '. ' || p.item_nome || ' (faltando)'
          WHEN 'manutencao' THEN '🔧 ' || p.item_numero || '. ' || p.item_nome || ' (manutenção)'
          ELSE '⚠️ ' || p.item_numero || '. ' || p.item_nome || ' (' || p.tipo_problema || ')'
        END, E'\\n' ORDER BY p.item_numero
      ) FROM problemas p) as problemas_texto,
      COALESCE((SELECT jsonb_agg(jsonb_build_object(
        'numero', p.item_numero, 'tipo', p.tipo_problema, 'descricao', p.descricao,
        'reportado_por_nome', (SELECT u.nome FROM hub_usuarios u WHERE u.id = (
          SELECT ep.reportado_por FROM eeg_problemas ep 
          WHERE UPPER(ep.caixa_codigo) = '${caixa}' AND ep.item_numero = p.item_numero 
            AND ep.status = 'pendente' ORDER BY ep.id DESC LIMIT 1)),
        'reportado_em', (SELECT TO_CHAR(ep.criado_em, 'DD/MM/YYYY HH24:MI') FROM eeg_problemas ep 
          WHERE UPPER(ep.caixa_codigo) = '${caixa}' AND ep.item_numero = p.item_numero 
            AND ep.status = 'pendente' ORDER BY ep.id DESC LIMIT 1),
        'ultima_nota', (SELECT pa.observacao FROM eeg_problema_acoes pa 
          JOIN eeg_problemas ep2 ON ep2.id = pa.problema_id
          WHERE UPPER(ep2.caixa_codigo) = '${caixa}' AND ep2.item_numero = p.item_numero 
            AND ep2.status = 'pendente' ORDER BY pa.criado_em DESC LIMIT 1)
      )) FROM problemas p), '[]'::jsonb) as problemas_json,
      n.texto as notas_supervisao,
      n.qtd as notas_quantidade
    FROM retirada r
    CROSS JOIN notas n
  `;
  
  respostaJSON = {
    mensagem: `📋 *RESUMO DA RETIRADA*\n\n👤 Técnico: ${userName}\n📅 Data: ${dataAtual}\n\n📦 Caixa: ${caixa}\n🩹 Cinta: _[CINTA_TEXTO]_\n📶 Modem: _Nenhum_\n\n*📦 Itens selecionados:*\n[LISTA_ITENS]\n\n*Confirma a retirada?*\n\n⚠️ *ATENÇÃO*\n_O registro correto dos equipamentos é sua responsabilidade. Informações inconsistentes serão auditadas e podem resultar em notificação à supervisão/gerência._`,
    botoes: [
      { id: `confirmar_retirada_${caixa}`, title: "✅ Confirmar" },
      { id: `cancelar_${caixa}`, title: "❌ Cancelar" },
      { id: "modulo_eeg", title: "🔙 Menu EEG" }

function handleDevolucaoIniciar(buttonId) {
  const caixaMatch = buttonId.match(/iniciar_devolucao_(C\d+)/i);
  const caixaCodigo = caixaMatch[1].toUpperCase();
  dadosExtraidos.caixa = caixaCodigo;
  
  executarSQLAntes = true;
  sqlAcaoAntes = 'verificar_exame_continuo_devolucao';
  sqlQueryAntes = `
    WITH exame_continuo AS (
      SELECT 
        e.id as exame_id,
        e.paciente_nome,
        e.hospital_nome,
        e.setor_leito,
        e.tipo_exame_nome,
        e.tecnico_id,
        e.is_continuo,
        e.status as exame_status,
        e.itens_com_paciente,
        u.nome as tecnico_nome
      FROM eeg_exames e
      LEFT JOIN hub_usuarios u ON u.id = e.tecnico_id
      WHERE e.caixa_codigo = '${caixaCodigo}'
        AND e.is_continuo = true
        AND e.status IN ('em_andamento', 'devolucao_parcial')
      ORDER BY e.id DESC
      LIMIT 1
    ),
    cleanup AS (
      UPDATE eeg_sessao_retirada 
      SET status = 'cancelada', atualizada_em = NOW()
      WHERE telefone LIKE '%' || RIGHT('${telefone}', 11) || '%'
        AND status NOT IN ('finalizada', 'cancelada', 'expirada')
    ),
    criar_sessao AS (
      INSERT INTO eeg_sessao_retirada (

        iniciada_em, atualizada_em, expira_em
      )
      VALUES (
        '${telefone}', ${userId}, 
        CASE WHEN (SELECT exame_id FROM exame_continuo) IS NOT NULL THEN 'devolucao_parcial' ELSE 'devolvendo' END,
        '${caixaCodigo}', 
        jsonb_build_object(
          'tipo_sessao', CASE WHEN (SELECT exame_id FROM exame_continuo) IS NOT NULL THEN 'devolucao_parcial' ELSE 'devolucao' END,
          'caixa_devolucao', '${caixaCodigo}',
          'exame_id', (SELECT exame_id FROM exame_continuo),
          'exame_continuo', (SELECT exame_id FROM exame_continuo) IS NOT NULL
        ),
        NOW(), NOW(), NOW() + INTERVAL '30 minutes'
      )
      RETURNING id
    )
    SELECT 
      ec.exame_id,
      ec.paciente_nome,
      ec.hospital_nome,
      ec.setor_leito,
      ec.tipo_exame_nome,
      ec.tecnico_nome,
      ec.is_continuo,
      ec.exame_status,
      ec.itens_com_paciente,
      CASE WHEN ec.exame_id IS NOT NULL THEN true ELSE false END as tem_exame_continuo
    FROM (SELECT 1) dummy

    mensagem: "📝 *Verificando caixa...*",

function handleTextoSelecaoEmprestimo() {
  // Dispatcher - verifica qual tipo de input é
  
  // 1. Número de item (1-13)
  if (content.match(/^\d+$/) && parseInt(content) >= 1 && parseInt(content) <= 13) {
    handleTextoNumeroItemEmprestimo();
    return;
  }
  
  // 2. Código de caixa (C2 ou EEG-C02-XXXXX)
  if (content.match(/^C\d+$/i) || content.match(/^EEG-C(\d+)-[A-Z0-9]{8}$/i)) {
    handleTextoCodigoCaixaEmprestimo();
    return;
  }
  
  // Não reconhecido
  respostaJSON = {
    mensagem: "🤔 Não entendi. Digite:\n• Número do item (1-13)\n• Código de caixa (ex: C5)",
    botoes: [
      { id: "eeg_emprestimo", title: "🔙 Voltar" },
      { id: "modulo_eeg", title: "🔙 Menu EEG" }
    ]
  };
}

function handleTextoCodigoCaixaEmprestimo() {
  let caixaOrigem = '';
  if (content.match(/^C\d+$/i)) {
    caixaOrigem = content.toUpperCase();
  } else {
    const qr = content.match(/^EEG-C(\d+)-[A-Z0-9]{8}$/i);
    caixaOrigem = 'C' + parseInt(qr[1]);
  }
  
  const caixaDestino = sessaoData?.problema_pendente?.caixa_destino || sessaoData?.caixa_codigo;
  
  tipoInput = 'codigo_caixa_emprestimo';
  dadosExtraidos.caixaOrigem = caixaOrigem;
  dadosExtraidos.caixaDestino = caixaDestino;
  
  // Não pode pegar da própria caixa
  if (caixaOrigem.toUpperCase() === caixaDestino?.toUpperCase()) {
    respostaJSON = {
      mensagem: `⚠️ *Caixa Inválida*\n\nVocê não pode pegar item emprestado da sua própria caixa (${caixaOrigem}).\n\nDigite outro código de caixa.`,
      botoes: [
        { id: "eeg_emprestimo", title: "🔙 Voltar" },
        { id: "modulo_eeg", title: "🔙 Menu EEG" }
      ]
    };
  } else {
    executarSQLAntes = true;
    sqlAcaoAntes = 'listar_itens_emprestimo_caixa';
    sqlQueryAntes = `
      WITH sessao AS (
        UPDATE eeg_sessao_retirada 
        SET problema_pendente = problema_pendente ||

function handleExameMeus() {
  executarSQLAntes = true;
  sqlAcaoAntes = 'listar_meus_exames_plantao';
  sqlQueryAntes = `
    WITH plantao_atual AS (
      SELECT id, inicio_em
      FROM eeg_plantoes
      WHERE tecnico_id = ${userId}
        AND status = 'ativo'
      ORDER BY id DESC
      LIMIT 1
    )
    SELECT
      e.id as exame_id,
      e.status,
      e.paciente_nome,
      e.hospital_nome,
      e.setor_leito,
      e.tipo_exame_nome,
      e.is_continuo,
      e.caixa_codigo,
      e.modem_codigo,
      e.itens_com_paciente,
      e.plantao_id,
      TO_CHAR(e.inicio_em, 'YYYY-MM-DD') as plantao_data,
      TO_CHAR(e.inicio_em, 'HH24:MI') as hora_inicio,
      TO_CHAR(e.fim_em, 'HH24:MI') as hora_fim,
      EXTRACT(EPOCH FROM (COALESCE(e.fim_em, NOW()) - e.inicio_em)) / 60 as duracao_minutos
    FROM eeg_exames e
    WHERE e.tecnico_id = ${userId}
      AND (
        e.status IN ('em_andamento', 'devolucao_parcial')
        OR (e.status = 'finalizado' AND e.plantao_id = (SELECT id FROM plantao_atual))
      )
    ORDER BY
      CASE e.status
        WHEN 'em_andamento' THEN 1
        WHEN 'devolucao_parcial' THEN 2
        ELSE 3
      END,
      e.inicio_em DESC
  `;

    handleTextoNaoReconhecido();

// Substituir o placeholder handleTextoExame()
function handleTextoExame() {
  const aguardando = sessaoData?.problema_pendente?.aguardando;
  
  if (aguardando === 'numero_hospital' && /^\d+$/.test(content.trim())) {
    handleTextoNumeroHospital();
  }
  else if (aguardando === 'numero_clinica' && /^\d+$/.test(content.trim())) {
    handleTextoNumeroClinica();
  }
  else if (aguardando === 'tipo_exame' && /^\d+$/.test(content.trim())) {
    handleTextoTipoExame();
  }
  else if (aguardando === 'endereco') {
    handleTextoEnderecoResidencial();
  }
  else if (aguardando === 'paciente') {
    handleTextoPacienteNome();
  }
  else if (aguardando === 'setor') {
    handleTextoSetorLeito();
  }
  else if (aguardando === 'itens_devolvendo' && /^[\d,\s]+$/.test(content)) {
    handleTextoItensDevolvendo();
  }
  else if (aguardando === 'itens_paciente' && /^[\d,\s]+$/.test(content)) {
    handleTextoItensPaciente();
  }
  else {
    respostaJSON = {
      mensagem: "🤔 Não entendi. Use o menu de exames:",
      botoes: [
        { id: "eeg_exames", title: "📋 Menu Exames" },
        { id: "modulo_eeg", title: "🔙 Menu EEG" }
      ]
    };
  }
}

function handleTextoNumeroHospital() {
  tipoInput = 'numero_hospital';
  const numeroDigitado = parseInt(content.trim());
  const regiao = sessaoData?.problema_pendente?.regiao || '';
  
  executarSQLAntes = true;
  sqlAcaoAntes = 'selecionar_hospital_por_numero';
  sqlQueryAntes = `
    WITH hospitais AS (
      SELECT 
        id, nome_curto, nome,
        ROW_NUMBER() OVER (ORDER BY ordem_exibicao, nome_curto) as numero
      FROM eeg_locais_parceiros 
      WHERE tipo = 'hospital' AND ativo = true 
        ${regiao ? `AND regiao = '${regiao}'` : ''}
    ),
    hospital_selecionado AS (
      SELECT id, nome_curto, nome FROM hospitais WHERE numero = ${numeroDigitado}
    ),
    salvar_local AS (
      UPDATE eeg_sessao_retirada 
      SET problema_pendente = problema_pendente || 
          jsonb_build_object(
            'local_id', (SELECT id FROM hospital_selecionado),
            'local_nome', (SELECT nome_curto FROM hospital_selecionado),
            'aguardando', 'continuo'

function handleRelStatusEquipamentos() {
  executarSQLAntes = true;
  sqlAcaoAntes = 'relatorio_status_equipamentos_v16';
  sqlQueryAntes = `
    WITH caixas_em_uso AS (
      SELECT 
        c.tecnico_atual_id as tecnico_id,
        u.nome as tecnico_nome,
        c.codigo as caixa_codigo,
        c.data_ultima_retirada,
        TO_CHAR(c.data_ultima_retirada, 'DD/MM HH24:MI') as retirada_formatada,
        ROUND(EXTRACT(EPOCH FROM (NOW() - c.data_ultima_retirada))/3600, 1) as horas_uso,
        13 as total_itens,
        13 - COALESCE((
          SELECT COUNT(DISTINCT item_numero) 
          FROM eeg_problemas pr 
          WHERE UPPER(pr.caixa_codigo) = UPPER(c.codigo) 
            AND pr.status = 'pendente'
        ), 0) as itens_ok,
        (
          SELECT e.hospital_nome
          FROM eeg_exames e
          WHERE UPPER(e.caixa_codigo) = UPPER(c.codigo)
            AND e.status IN ('em_andamento', 'devolucao_parcial')
          ORDER BY e.id DESC
          LIMIT 1
        ) as hospital
      FROM eeg_caixas c
      JOIN hub_usuarios u ON u.id = c.tecnico_atual_id

function handleLocalizacao() {
  tipoInput = 'localizacao';
  dadosExtraidos.latitude = latitude;
  dadosExtraidos.longitude = longitude;
  
  const tipoSessaoBanco = sessaoData?.problema_pendente?.tipo_sessao;
  const sessaoStatusLoc = sessaoData?.sessao_status || sessaoData?.status;
  
  // ───────────────────────────────────────────────────────────────────
  // LOCALIZAÇÃO PARA INICIAR PLANTÃO
  // ───────────────────────────────────────────────────────────────────
  if (tipoSessaoBanco === 'inicio_plantao' || tipoSessaoBanco === 'inicio_plantao_com_caixa') {
    handleLocInicioPlantao();
    return;
  }
  
  // ───────────────────────────────────────────────────────────────────
  // LOCALIZAÇÃO PARA FINALIZAR PLANTÃO (com itens)
  // ───────────────────────────────────────────────────────────────────
  if (tipoSessaoBanco === 'finalizar_plantao') {
    handleLocFinalizarPlantao();
    return;
  }
  
  // ───────────────────────────────────────────────────────────────────
  // LOCALIZAÇÃO PARA DEVOLUÇÃO PARCIAL
  // ───────────────────────────────────────────────────────────────────
  if (tipoSessaoBanco === 'devolucao_parcial') {
    handleLocDevolucaoParcial();
    return;
  }
  
  // ───────────────────────────────────────────────────────────────────
  // LOCALIZAÇÃO PARA ASSUMIR DEVOLUÇÃO
  // ───────────────────────────────────────────────────────────────────
  if (tipoSessaoBanco === 'assumir_devolucao') {
    handleLocAssumirDevolucao();
    return;
  }
  
  // ───────────────────────────────────────────────────────────────────
  // v16: LOCALIZAÇÃO PARA RETIRADA VIA QR DIRETO
  // ───────────────────────────────────────────────────────────────────
  if (tipoSessaoBanco === 'retirada_qr_direto') {
    handleLocRetiradaQRDireto();
    return;
  }
  
  // ───────────────────────────────────────────────────────────────────
  // LOCALIZAÇÃO GENÉRICA (retirada/devolução padrão)
  // ───────────────────────────────────────────────────────────────────
  handleLocGenerica();
}

// v16: LOCALIZAÇÃO RECEBIDA VIA QR DIRETO → IR PARA SELEÇÃO DE ITENS
// ───────────────────────────────────────────────────────────────────
function handleLocRetiradaQRDireto() {
  const caixaPendente = sessaoData?.problema_pendente?.caixa_pendente || sessaoData?.caixa_codigo;
  const caixaRestritaFlag = sessaoData?.problema_pendente?.caixa_restrita;
  
  if (!caixaPendente) {
    respostaJSON = {
      mensagem: `📍 *Localização registrada!*\n\n📦 Agora escaneie o QR Code da caixa\nou digite o código completo.\n\n⚠️ Algumas caixas só podem ser retiradas na clínica.`,
      botoes: [
        { id: "modulo_eeg", title: "🔙 Menu EEG" }
      ]
    };
    return;
  }
  
  const CAIXAS_RESTRITAS = ['C1','C2','C3','C4','C5','C6','C9','C12','C14','C15','C16'];
  const caixaRestrita = caixaRestritaFlag === true || caixaRestritaFlag === 't' || CAIXAS_RESTRITAS.includes(caixaPendente);
  
  // Coordenadas da clínica
  const CLINICA_LAT = -15.8280632;
  const CLINICA_LNG = -47.9299076;
  
  executarSQLAntes = true;
  sqlAcaoAntes = 'loc_qr_direto_selecao_itens';
  sqlQueryAntes = `
    WITH caixa_status AS (
      SELECT 
        c.id,
        c.codigo,
        c.status,
        c.tecnico_atual_id,
        u.nome as tecnico_nome,
        CASE 
          WHEN c.status = 'em_uso' AND c.tecnico_atual_id = ${userId} THEN 'minha'
          WHEN c.status = 'em_uso' AND c.tecnico_atual_id IS NOT NULL AND c.tecnico_atual_id != ${userId} THEN 'outro'
          WHEN c.bloqueada = true THEN 'bloqueada'
          ELSE 'disponivel'
        END as situacao
      FROM eeg_caixas c
      LEFT JOIN hub_usuarios u ON u.id = c.tecnico_atual_id
      WHERE UPPER(c.codigo) = UPPER('${caixaPendente}')
    ),
    registra_loc AS (
      UPDATE eeg_sessao_retirada 
      SET latitude = ${latitude}, 
          longitude = ${longitude},
          localizacao_registrada_em = NOW(),
          localizacao_dentro_raio = calcular_distancia_metros(${latitude}, ${longitude}, ${CLINICA_LAT}, ${CLINICA_LNG}) <= 100,
          localizacao_distancia = calcular_distancia_metros(${latitude}, ${longitude}, ${CLINICA_LAT}, ${CLINICA_LNG}),
          status = CASE 
            WHEN (SELECT situacao FROM caixa_status) = 'disponivel' THEN 'selecionando'
            ELSE 'cancelada'
          END,
          atualizada_em = NOW(),
          expira_em = NOW() + INTERVAL '30 minutes'
      WHERE id = (
      SELECT id FROM eeg_sessao_retirada
         WHERE telefone LIKE '%' || RIGHT('${telefone}', 11) || '%'
         AND status NOT IN ('finalizada', 'cancelada', 'expirada')
         ORDER BY id DESC LIMIT 1
    )
      RETURNING id, localizacao_dentro_raio, localizacao_distancia
    ),
    itens AS (
      SELECT * FROM buscar_itens_caixa_com_status_v16('${caixaPendente}')
    )
    SELECT 
      cs.situacao,
      cs.tecnico_nome,
      r.id as sessao_id,
      r.localizacao_dentro_raio as dentro_raio,
      r.localizacao_distancia as distancia,
      i.lista_itens_texto,
      i.lista_itens_json,
      i.tem_problemas,
      i.problemas_quantidade,
      i.notas_supervisao,
      i.notas_quantidade,
      i.exame_continuo_info,
      i.total_indisponiveis,
      '${caixaPendente}' as caixa_codigo,
      ${caixaRestrita} as caixa_restrita
    FROM caixa_status cs, registra_loc r, itens i

// [INPUT] QR CODE — SUBSTITUIR handleQRCode
// ═══════════════════════════════════════════════════════════════

function handleQRCode(qrMatch) {
  tipoInput = 'qr_code';
  const caixaCodigo = 'C' + parseInt(qrMatch[1]);
  dadosExtraidos.caixa = caixaCodigo;
  dadosExtraidos.qrCode = content.toUpperCase();
  
  const tipoSessaoBanco = sessaoData?.problema_pendente?.tipo_sessao;
  const caixaDevolucao = sessaoData?.problema_pendente?.caixa_devolucao;
  const sessaoStatus = sessaoData?.sessao_status || sessaoData?.status;
  
  const sessaoAtiva = sessaoData && 
                      sessaoStatus && 
                      !['finalizada', 'cancelada', 'expirada'].includes(sessaoStatus);

  const caixaSessaoAtual = sessaoData?.caixa_codigo || '';
  const sessaoMesmaCaixa = !caixaSessaoAtual || 
                           caixaSessaoAtual.toUpperCase() === caixaCodigo.toUpperCase();

  const ehDevolucaoSessao = sessaoAtiva && 
                      ((tipoSessaoBanco === 'devolucao') || (sessaoStatus === 'devolvendo'));

  const sessaoJaTemLocalizacao = sessaoData?.localizacao_registrada_em || 
                                  sessaoData?.latitude ||
                                  sessaoStatus === 'selecionando' ||
                                  sessaoStatus === 'aguardando_modem' ||
                                  sessaoStatus === 'aguardando_qr';

  const tiposRetirada = ['retirada', 'retirada_qr_direto', 'inicio_plantao', 'inicio_plantao_com_caixa'];
  const ehTipoRetirada = tiposRetirada.includes(tipoSessaoBanco);

  const ehRetirada = sessaoAtiva && 
                     sessaoMesmaCaixa &&
                     ehTipoRetirada &&
                     sessaoJaTemLocalizacao;

  const caixaRestrita = CAIXAS_RESTRITAS_CLINICA.includes(caixaCodigo);

  // ─────────────────────────────────────────────────────────────────
  // CENÁRIO 0: EMPRÉSTIMO
  // ─────────────────────────────────────────────────────────────────
  const ehEmprestimo = sessaoAtiva && 
                       (tipoSessaoBanco === 'emprestimo' || 
                        sessaoStatus === 'selecionando_caixa_emprestimo');
  
  if (ehEmprestimo) {
    handleQRCenarioEmprestimo(caixaCodigo);
    return;
  }
  
  // ─────────────────────────────────────────────────────────────────
  // CENÁRIO 1: DEVOLUÇÃO (sessão já ativa)
  // ─────────────────────────────────────────────────────────────────
  if (ehDevolucaoSessao && caixaDevolucao) {
    handleQRCenarioDevolucao(caixaCodigo, caixaDevolucao);
    return;
  }
  
  // ─────────────────────────────────────────────────────────────────
  // CENÁRIO 2: SEM PERMISSÃO
  // ─────────────────────────────────────────────────────────────────
  if (!podeRetirar) {
    respostaJSON = {
      mensagem: "⚠️ *Sem Permissão*\n\nVocê não tem permissão para retirar equipamentos.",
      botoes: [
        { id: "menu_principal", title: "🏠 Menu" },
        { id: "modulo_eeg", title: "🔙 Menu EEG" }
      ]
    };
    return;
  }
  
  // ─────────────────────────────────────────────────────────────────
  // CENÁRIO 3: RETIRADA COM LOCALIZAÇÃO (sessão ativa)
  // ─────────────────────────────────────────────────────────────────
  if (ehRetirada) {
    handleQRCenarioRetiradaComLoc(caixaCodigo, caixaRestrita);
    return;
  }
  
  // ─────────────────────────────────────────────────────────────────
  // CENÁRIO 4: SESSÃO ATIVA SEM LOCALIZAÇÃO
  // ─────────────────────────────────────────────────────────────────
  if (sessaoAtiva && ehTipoRetirada && !sessaoJaTemLocalizacao) {
    handleQRCenarioRetiradaSemLoc(caixaCodigo, caixaRestrita);
    return;
  }
  
  // ─────────────────────────────────────────────────────────────────
  // CENÁRIO 5: SEM SESSÃO - VERIFICAR SE É MINHA CAIXA OU NOVA
  // ─────────────────────────────────────────────────────────────────
  handleQRCenarioVerificarCaixa(caixaCodigo, caixaRestrita);
}

function handleQRCenarioNovaSessao(caixaCodigo, caixaRestrita) {
  executarSQLAntes = true;
  sqlAcaoAntes = 'verificar_caixa_qr_inteligente';
  sqlQueryAntes = `
    WITH 
    cancela_antigas AS (
      UPDATE eeg_sessao_retirada 
      SET status = 'cancelada', atualizada_em = NOW()
      WHERE telefone LIKE '%' || RIGHT('${telefone}', 11) || '%'
        AND status NOT IN ('finalizada', 'cancelada', 'expirada')
        AND (problema_pendente IS NULL OR problema_pendente->>'tipo_sessao' NOT IN ('devolucao', 'exame'))
      RETURNING id as cancelada_id
    ),
    caixa_info AS (
      SELECT 
        c.id as caixa_id,
        c.codigo,
        c.status as caixa_status,
        c.tecnico_atual_id,
        u.nome as tecnico_nome,
        -- É minha caixa?
        CASE WHEN c.tecnico_atual_id = ${userId} THEN true ELSE false END as eh_minha_caixa,
        -- Caixa disponível para retirada?
        CASE 
          WHEN c.status = 'em_uso' AND c.tecnico_atual_id IS NOT NULL 
            AND c.tecnico_atual_id != ${userId} THEN false
          ELSE true
        END as disponivel
      FROM eeg_caixas c
      LEFT JOIN hub_usuarios u ON u.id = c.tecnico_atual_id
      WHERE UPPER(c.codigo) = UPPER('${caixaCodigo}')
    ),
    exame_ativo AS (
      -- Exame em andamento com esta caixa
      SELECT 
        e.id as exame_id,
        e.paciente_nome,
        e.hospital_nome,
        e.setor_leito,
        e.tipo_exame_nome,
        e.is_continuo,
        e.status as exame_status,
        TO_CHAR(e.inicio_em, 'DD/MM HH24:MI') as inicio_formatado,
        ROUND(EXTRACT(EPOCH FROM (NOW() - e.inicio_em)) / 60) as duracao_minutos
      FROM eeg_exames e
      WHERE UPPER(e.caixa_codigo) = UPPER('${caixaCodigo}')
        AND e.status IN ('em_andamento', 'devolucao_parcial')
      ORDER BY e.inicio_em DESC
      LIMIT 1
    ),
    exame_parcial AS (
      -- Exame com devolução parcial (itens com paciente)
      SELECT 
        e.id as exame_id,
        e.paciente_nome,
        e.hospital_nome,
        e.setor_leito,
        e.itens_com_paciente,
        u.nome as instalador_nome,
        TO_CHAR(e.parcial_em, 'DD/MM HH24:MI') as parcial_formatado
      FROM eeg_exames e
      JOIN hub_usuarios u ON u.id = e.tecnico_id
      WHERE UPPER(e.caixa_codigo) = UPPER('${caixaCodigo}')
        AND e.status = 'devolucao_parcial'
      ORDER BY e.parcial_em DESC
      LIMIT 1
    ),
    itens_emprestados AS (
      -- Itens emprestados desta caixa
      SELECT 
        COUNT(*) as qtd_emprestados,
        string_agg(ip.nome, ', ') as nomes_emprestados
      FROM eeg_emprestimos emp
      JOIN eeg_itens_padrao ip ON ip.numero::VARCHAR = emp.item_codigo
      WHERE emp.caixa_origem = UPPER('${caixaCodigo}')
        AND emp.status = 'ativo'
    ),
    itens_problema AS (
      -- Itens com problema pendente
      SELECT 
        COUNT(*) as qtd_problemas,
        string_agg(ip.nome, ', ') as nomes_problemas
      FROM eeg_problemas p
      JOIN eeg_itens_padrao ip ON ip.numero = p.item_numero
      WHERE UPPER(p.caixa_codigo) = UPPER('${caixaCodigo}')
        AND p.status = 'pendente'
    ),
    nova_sessao AS (
      INSERT INTO eeg_sessao_retirada (
        telefone, usuario_id, caixa_codigo, status, problema_pendente, 
        iniciada_em, atualizada_em, expira_em
      )
      SELECT 
        '${telefone}', ${userId}, '${caixaCodigo}',
        CASE 
          WHEN ci.eh_minha_caixa THEN 'devolvendo'
          ELSE 'aguardando_localizacao'
        END,
        jsonb_build_object(
          'tipo_sessao', CASE WHEN ci.eh_minha_caixa THEN 'devolucao' ELSE 'retirada_qr_direto' END,
          'caixa_pendente', '${caixaCodigo}',
          'caixa_restrita', ${caixaRestrita},
          'exame_parcial_id', (SELECT exame_id FROM exame_parcial)
        ),
        NOW(), NOW(), NOW() + INTERVAL '30 minutes'
      FROM caixa_info ci
      WHERE ci.disponivel = true OR ci.eh_minha_caixa = true
      RETURNING id as sessao_id
    )
    SELECT 
      ci.caixa_id, 
      ci.codigo, 
      ci.caixa_status, 
      ci.tecnico_atual_id,
      ci.tecnico_nome, 
      ci.disponivel, 
      ci.eh_minha_caixa, 
      ns.sessao_id,
      '${caixaCodigo}' as caixa_pendente, 
      ${caixaRestrita} as caixa_restrita,
      -- Exame ativo
      ea.exame_id as exame_ativo_id,
      ea.paciente_nome as exame_paciente,
      ea.hospital_nome as exame_hospital,
      ea.setor_leito as exame_setor,
      ea.tipo_exame_nome as exame_tipo,
      ea.is_continuo as exame_continuo,
      ea.exame_status,
      ea.inicio_formatado as exame_inicio,
      ea.duracao_minutos as exame_duracao,
      -- Exame parcial
      ep.exame_id as exame_parcial_id,
      ep.itens_com_paciente as exame_itens_paciente,
      ep.instalador_nome as exame_instalador,
      ep.parcial_formatado as exame_parcial_data,
      -- Empréstimos
      COALESCE(ie.qtd_emprestados, 0) as qtd_emprestados,
      ie.nomes_emprestados,
      -- Problemas
      COALESCE(ipr.qtd_problemas, 0) as qtd_problemas,
      ipr.nomes_problemas
    FROM caixa_info ci
    LEFT JOIN nova_sessao ns ON true
    LEFT JOIN exame_ativo ea ON true
    LEFT JOIN exame_parcial ep ON true
    LEFT JOIN itens_emprestados ie ON true
    LEFT JOIN itens_problema ipr ON true
  `;
  
  respostaJSON = {
    mensagem: `📦 *Verificando caixa ${caixaCodigo}...*`,
    botoes: [{ id: "modulo_eeg", title: "🔙 Menu EEG" }],
    _verificarCaixaQRInteligente: true,
    _caixaPendente: caixaCodigo,
    _caixaRestrita: caixaRestrita,
    _telefone: telefone,
    _userId: userId
  };
}

function handleQRCenarioVerificarCaixa(caixaCodigo, caixaRestrita) {
  // Verificar no banco se a caixa é do técnico ou disponível
  // E JÁ CRIAR SESSÃO COM A CAIXA PENDENTE
  executarSQLAntes = true;
  sqlAcaoAntes = 'verificar_caixa_qr_direto';
  sqlQueryAntes = `
    WITH cancela_sessao_anterior AS (
      UPDATE eeg_sessao_retirada
      SET status = 'cancelada', atualizada_em = NOW()
      WHERE telefone LIKE '%' || RIGHT('${telefone}', 11) || '%'
        AND status NOT IN ('finalizada', 'cancelada', 'expirada')
        AND (problema_pendente IS NULL 
             OR problema_pendente->>'tipo_sessao' IN ('retirada', 'retirada_qr_direto'))
    ),
    cria_sessao_qr AS (
      INSERT INTO eeg_sessao_retirada (
        telefone, usuario_id, caixa_codigo, status, problema_pendente, 
        iniciada_em, atualizada_em, expira_em
      ) VALUES (
        '${telefone}', ${userId}, '${caixaCodigo}', 'aguardando_localizacao',
        jsonb_build_object(
          'tipo_sessao', 'retirada_qr_direto',
          'caixa_pendente', '${caixaCodigo}',
          'caixa_restrita', ${caixaRestrita}
        ),
        NOW(), NOW(), NOW() + INTERVAL '30 minutes'
      )
      RETURNING id
    )
    SELECT 
      c.id as caixa_id,
      c.codigo,
      c.status as caixa_status,
      c.tecnico_atual_id,
      u.nome as tecnico_nome,
      (SELECT id FROM cria_sessao_qr) as sessao_criada_id,
      CASE 
        WHEN c.status = 'em_uso' AND c.tecnico_atual_id = ${userId} THEN 'minha'
        WHEN c.status = 'em_uso' AND c.tecnico_atual_id IS NOT NULL AND c.tecnico_atual_id != ${userId} THEN 'outro'
        ELSE 'disponivel'
      END as situacao,
      -- Exame ativo (do próprio técnico)
      (
        SELECT jsonb_build_object(
          'exame_id', e.id,
          'paciente_nome', e.paciente_nome,
          'hospital_nome', e.hospital_nome,
          'is_continuo', e.is_continuo
        )
        FROM eeg_exames e
        WHERE UPPER(e.caixa_codigo) = UPPER('${caixaCodigo}')
          AND e.status = 'em_andamento'
          AND e.tecnico_id = ${userId}
        LIMIT 1
      ) as exame_ativo,
      -- Exame parcial (devolução pendente)
      (
        SELECT jsonb_build_object(
          'exame_id', e.id,
          'paciente_nome', e.paciente_nome,
          'hospital_nome', e.hospital_nome,
          'setor_leito', e.setor_leito,
          'itens_com_paciente', e.itens_com_paciente
        )
        FROM eeg_exames e
        WHERE UPPER(e.caixa_codigo) = UPPER('${caixaCodigo}')
          AND e.status = 'devolucao_parcial'
        LIMIT 1
      ) as exame_parcial,
      -- Contadores
      COALESCE((
        SELECT COUNT(*)::INTEGER
        FROM eeg_problemas p
        WHERE UPPER(p.caixa_codigo) = UPPER('${caixaCodigo}')
          AND p.status NOT IN ('resolvido', 'cancelado')
      ), 0) as qtd_problemas,
      COALESCE((
        SELECT COUNT(*)::INTEGER
        FROM eeg_emprestimos e
        WHERE e.caixa_origem = UPPER('${caixaCodigo}')
          AND e.status = 'ativo'
      ), 0) as qtd_emprestados,
      '${caixaCodigo}' as caixa_codigo_param,
      ${caixaRestrita} as caixa_restrita
    FROM eeg_caixas c
    LEFT JOIN hub_usuarios u ON u.id = c.tecnico_atual_id
    WHERE UPPER(c.codigo) = UPPER('${caixaCodigo}')
  `;
  
  respostaJSON = {
    mensagem: `📦 *Verificando caixa ${caixaCodigo}...*`,
    botoes: [{ id: "modulo_eeg", title: "🔙 Menu EEG" }],
    _verificarCaixaQRDireto: true,
    _caixaCodigo: caixaCodigo,
    _caixaRestrita: caixaRestrita,
    _telefone: telefone,
    _userId: userId
  };
}

// Substitui TODAS as versões anteriores (Etapas 2, 3, 4)

function handleTextoContexto() {
  const tipoSessao = sessaoData?.problema_pendente?.tipo_sessao;
  const aguardando = sessaoData?.problema_pendente?.aguardando;
  const sessaoStatus = sessaoData?.status;
  const origem = sessaoData?.problema_pendente?.origem;
  
  // Verificar se parece seleção numérica
  const pareceSelecao = content && /^[\d,\s]+$/.test(content.trim());
  
  // ─────────────────────────────────────────────────────────────
  // [F7] PLANTÃO
  // ─────────────────────────────────────────────────────────────
  
  // Motivo outro (manter caixa simples)
  if (aguardando === 'motivo_outro') {
    handleTextoMotivoOutroPlantao();
    return;
  }
  
  // Motivo outro (manter itens específicos)
  if (tipoSessao === 'finalizar_plantao_outro') {
    handleTextoMotivoOutroItens();
    return;
  }
  
  // ─────────────────────────────────────────────────────────────
  // [F1] RETIRADA
  // ─────────────────────────────────────────────────────────────
  
  // Seleção de itens (status 'selecionando')
  if (sessaoStatus === 'selecionando' && pareceSelecao) {
    handleSelecaoItensRetirada();
    return;
  }
  
  // Seleção de cintas (status 'selecionando_cinta')
  if (sessaoStatus === 'selecionando_cinta' && 
      pareceSelecao &&
      tipoSessao !== 'emprestimo' &&
      tipoSessao !== 'exame') {
    handleSelecaoCintas();
    return;
  }
  
  // Seleção de modems (status 'selecionando_modem')
  if (sessaoStatus === 'selecionando_modem' && 
      pareceSelecao &&
      tipoSessao !== 'emprestimo' &&
      tipoSessao !== 'exame') {
    handleSelecaoModems();
    return;
  }
  
  // Seleção de item com problema (status 'aguardando_problemas')
  if (sessaoStatus === 'aguardando_problemas' && 
      tipoSessao === 'devolucao' &&
      pareceSelecao) {
    handleTextoItemDevolucao();
    return;
  }
  
  // Descrição do problema (origem = 'devolucao' + item_numero + tipo_problema)
  if (origem === 'devolucao' && 
      sessaoData?.problema_pendente?.item_numero &&
      sessaoData?.problema_pendente?.tipo_problema &&
      content.length >= 3) {
    handleTextoDescProblemaDevolucao();
    return;
  }
  
  // ─────────────────────────────────────────────────────────────
  // [F4] EMPRÉSTIMO (TODO: Etapa 5)
  // ─────────────────────────────────────────────────────────────
  
  // Seleção de item para empréstimo
  if (sessaoStatus === 'selecionando_item_emprestimo' && pareceSelecao) {
    handleTextoSelecaoEmprestimo();
    return;
  }
  
  // ─────────────────────────────────────────────────────────────
  // [F8] EXAMES (TODO: Etapa 7)
  // ─────────────────────────────────────────────────────────────
  
  // Dados de exame (nome paciente, setor, etc.)
  if ((tipoSessao === 'exame' || tipoSessao === 'devolucao_parcial') && aguardando) {
    handleTextoExame();
    return;
  }
  
  // Seleção de modems para exame
  if (sessaoStatus === 'selecionando_modem' && 
      tipoSessao === 'exame' && 
      pareceSelecao) {
    handleTextoModemsExame();
    return;
  }
  
  // ─────────────────────────────────────────────────────────────
  // [F3] REPORTAR PROBLEMA (TODO: Etapa 8)
  // ─────────────────────────────────────────────────────────────
  
  // Descrição do problema reportado
  if (origem === 'reportar' && 
      sessaoData?.problema_pendente?.item_numero &&
      sessaoData?.problema_pendente?.tipo_problema &&
      content.length >= 3) {
    handleTextoDescProblemaReportar();
    return;
  }
  
  // ─────────────────────────────────────────────────────────────
  // [F5] MANUTENÇÃO (TODO: Etapa 9)
  // ─────────────────────────────────────────────────────────────
  
  // Observação de ação de manutenção
  if (tipoSessao === 'manutencao_acao' && content.length >= 3) {
    handleTextoObsManutencao();
    return;
  }
  
  // Nota de supervisão
  if (aguardando === 'nota_supervisao' && content.length >= 3) {
    handleTextoNotaSupervisao();
    return;
  }
 
// Observação de recuperação de item
  if (tipoSessao === 'resolver_recuperado' && content.length >= 3) {
    handleTextoObsRecuperado();
    return;
  }
  
  // Observação de substituição de item
  if (tipoSessao === 'resolver_substituido' && content.length >= 3) {
    handleTextoObsSubstituido();
    return;
  }
  
  // Código do novo item (substituição)
  if (tipoSessao === 'resolver_codigo_novo' && content.length >= 2) {
    handleTextoCodigoNovoItem();
    return;
  }
 
  // ─────────────────────────────────────────────────────────────
  // FALLBACK — texto não reconhecido no contexto
  // ─────────────────────────────────────────────────────────────
  tipoInput = 'texto_livre';
  respostaJSON = {
    mensagem: `🤔 Não entendi "*${content.substring(0, 30)}...*".\n\nSelecione uma opção:`,
    botoes: [
      { id: "modulo_eeg", title: "🧠 Módulo EEG" },
      { id: "menu_principal", title: "🏠 Menu" }
    ]
  };
}
